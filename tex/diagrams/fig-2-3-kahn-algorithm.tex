\begin{tikzpicture}[
    nd/.style={circle, draw, thick, minimum size=0.7cm, fill=white, font=\small, inner sep=0pt},
    active/.style={nd, fill=green!20, thick},
    emitted/.style={nd, opacity=0.25},
    arr/.style={->, thick, shorten >=1pt, shorten <=1pt},
    arrfaded/.style={arr, opacity=0.25},
    steplabel/.style={font=\small\bfseries, anchor=north},
    stepdesc/.style={font=\small, anchor=north, text width=5.2cm, align=center,

% Graph used throughout (deliberately scrambled IDs to show Kahn orders by
% dependency, not by id):
%   Layer 1 (in-degree 0): {3, 2, 5}
%   Layer 2:               {1, 6}    -- depend on layer 1
%   Layer 3:               {4}       -- depends on layer 2
% Edges: 3->1, 2->1, 3->6, 5->6, 1->4, 6->4

% =============== Step 1: frontier {3, 2, 5} ===============
\begin{scope}[local bounding box=s1]
    \node[active] (a3) at (0, 0)   {3};
    \node[active] (a2) at (0,-1)   {2};
    \node[active] (a5) at (0,-2)   {5};
    \node[nd]     (a1) at (1.6,-0.5) {1};
    \node[nd]     (a6) at (1.6,-1.5) {6};
    \node[nd]     (a4) at (3.2,-1)   {4};
    \draw[arr] (a3) -- (a1); \draw[arr] (a2) -- (a1);
    \draw[arr] (a3) -- (a6); \draw[arr] (a5) -- (a6);
    \draw[arr] (a1) -- (a4); \draw[arr] (a6) -- (a4);
\end{scope}
\node[steplabel] at ($(s1.south) + (0,-0.20)$) (l1) {Step 1};
\node[stepdesc]  at ($(l1.south) + (0,-0.05)$) {In-degree $0$: $\{3,\,2,\,5\}$.\\Emit: $\{3,\,2,\,5\}$.\\In-degree of $1, 6$ drops to $0$.};

% =============== Step 2: frontier {1, 6} ===============
\begin{scope}[shift={(5.8,0)}, local bounding box=s2]

    \node[emitted] (b5) at (0,-2)   {5};
